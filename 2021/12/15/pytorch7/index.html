Pytorch学习笔记7——一个深度学习程序的通用框架

以下是一个深度学习程序的通用框架，大部分基础的深度学习任务都可以按照以下这个框架的步骤去进行书写：

Step0: 一些依赖项函数

xxxxxxxxxx
def get_device():
    ''' Get device (if GPU is available, use GPU) '''
    return 'cuda' if torch.cuda.is_available() else 'cpu'
  
 # fix random seed
def same_seeds(seed):
    torch.manual_seed(seed)
    if torch.cuda.is_available():
        torch.cuda.manual_seed(seed)
        torch.cuda.manual_seed_all(seed)  
    np.random.seed(seed)  
    torch.backends.cudnn.benchmark = False
    torch.backends.cudnn.deterministic = True

Step1: 下载数据、准备数据（这一步将硬盘的数据加载到内存中）

xxxxxxxxxx
import numpy as np
print('Loading data ...')
# 加载Train Features、Train Labels、Test Features
train = np.load('train.npy')
train_label = np.load('train_label.npy')
test = np.load('test.npy')
# 一般而言可以打印一下数据集的大小，心中有数
print('Size of training data: {}'.format(train.shape))
print('Size of testing data: {}'.format(test.shape))

Step2: 创建Dataset类

xxxxxxxxxx
import torch
from torch.utils.data import Dataset
class MyDataset(Dataset):
    def __init__(self, X, y=None):   # 主要就是在init函数里定义好 self.data 和 self.label
        self.data = torch.from_numpy(X).float()  # 进行赋值
        if y is not None:   # 这儿需要考虑有y的训练集 和 没有y的测试集
            self.label = torch.LongTensor(y.astype(np.int))
        else:
            self.label = None


    def __getitem__(self, idx):    # 基本都是如下这个形式，不太需要变化
        if self.label is not None:
            return self.data[idx], self.label[idx]
        else:
            return self.data[idx]


    def __len__(self):  # 返回数据集长度，一般都是如下这个形式，也不太需要变化
        return len(self.data)

Step3: 将训练数据分割为 训练集 和 验证集

xxxxxxxxxx
VAL_RATIO = 0.2  # 利用此参数控制 验证集占 整个训练数据的比例


percent = int(train.shape[0] * (1 - VAL_RATIO))
train_x, train_y, val_x, val_y = train[:percent], train_label[:percent], train[percent:], train_label[percent:]


print('Size of training set: {}'.format(train_x.shape))
print('Size of validation set: {}'.format(val_x.shape))

Step4: 实例化Dataset对象以及对应的DataLoader对象

xxxxxxxxxx
BATCH_SIZE = 64  # 设定 Batch_size，此参数DataLoader


from torch.utils.data import DataLoader


train_set = MyDataset(train_x, train_y)
val_set = MyDataset(val_x, val_y)


#only shuffle the training data
train_loader = DataLoader(train_set, batch_size=BATCH_SIZE, shuffle=True) 
val_loader = DataLoader(val_set, batch_size=BATCH_SIZE, shuffle=False)

Step5: 创建网络模型

xxxxxxxxxx
import torch
import torch.nn as nn


class Classifier(nn.Module):
    def __init__(self):   # 定义网络层级
        super(Classifier, self).__init__()
        self.layer1 = nn.Linear(429, 1024)
        self.layer2 = nn.Linear(1024, 512)
        self.layer3 = nn.Linear(512, 128)
        self.out = nn.Linear(128, 39) 
        self.act_fn = nn.Sigmoid()
        
    def forward(self, x): # 定义数据的流转逻辑过程
        x = self.layer1(x)
        x = self.act_fn(x)
        
        x = self.layer2(x)
        x = self.act_fn(x)


        x = self.layer3(x)
        x = self.act_fn(x)


        x = self.out(x)
        
        return x

Step6: 设定一些训练参数，为训练作准备

xxxxxxxxxx
# fix random seed for reproducibility
same_seeds(0)


# get device 
device = get_device()
print(f'DEVICE: {device}')


# training parameters 
num_epoch = 20               # number of training epoch
learning_rate = 0.0001       # learning rate


# the path where checkpoint saved
model_path = './model.ckpt'


# create model, define a loss function, and optimizer
model = Classifier().to(device)
criterion = nn.CrossEntropyLoss() 
optimizer = torch.optim.Adam(model.parameters(), lr=learning_rate)

Step7: 开启网络训练，参数优化

xxxxxxxxxx
# start training


best_acc = 0.0
for epoch in range(num_epoch):
    train_loss = 0.0
    val_loss = 0.0


    # training
    model.train() # set the model to training mode
    for i, data in enumerate(train_loader):
        inputs, labels = data  # 从data中加载inputs和labels
        inputs, labels = inputs.to(device), labels.to(device) # 拷贝至设备
        optimizer.zero_grad()  # 清零梯度
        outputs = model(inputs)   # 将输入放到模型中拿到输出
        batch_loss = criterion(outputs, labels) # 通过损失函数，计算本次损失值
        batch_loss.backward()  # 损失反向传播，计算梯度
        optimizer.step() # 使用optimizer更新一步模型


        train_loss += batch_loss.item()


    # 每一个epoch完成后，如果有验证集的话，就进行validation验证
    if len(val_set) > 0:
        model.eval() # set the model to evaluation mode
        with torch.no_grad(): # 验证集不需要计算梯度
            for i, data in enumerate(val_loader):
                inputs, labels = data 
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs)
                batch_loss = criterion(outputs, labels) 


                val_loss += batch_loss.item()


            print('[{:03d}/{:03d}] Train Loss: {:3.6f} | Val Loss: {:3.6f}'.format(
                epoch + 1, num_epoch, train_loss/len(train_loader), val_loss/len(val_loader)
            ))


            # 必备过程，如果模型比截止当前更优了，那么就保存并记录，然后输出
            if val_acc > best_acc:
                best_acc = val_acc
                torch.save(model.state_dict(), model_path)
                print('saving model with acc {:.3f}'.format(best_acc/len(val_set)))
    else: # 没有验证集这一步的话，就可以直接打印输出
        print('[{:03d}/{:03d}] Train Acc: {:3.6f} Loss: {:3.6f}'.format(
            epoch + 1, num_epoch, train_acc/len(train_set), train_loss/len(train_loader)
        ))
        
# 如果没有验证这一步骤的话，我们需要保存最终训练得到的这个模型
if len(val_set) == 0:
    torch.save(model.state_dict(), model_path)
    print('saving model at last epoch')
